FILE(GLOB PIDX_GENERIC_RST_SOURCES *.h *.c)


# ////////////////////////////////////////
# includes
# ////////////////////////////////////////

IF (MPI_C_FOUND)
  #MESSAGE("Configuring pidx with MPI support")
  INCLUDE_DIRECTORIES(${MPI_C_INCLUDE_PATH})
  SET(PIDX_GENERIC_RST_LINK_LIBS ${PIDX_GENERIC_RST_LINK_LIBS} ${MPI_C_LIBRARIES})
ENDIF()

IF (ZFP_FOUND)
  #MESSAGE("Configuring pidx with ZFP support")
  INCLUDE_DIRECTORIES(${ZFP_INCLUDE_DIR})
  SET(PIDX_GENERIC_RST_LINK_LIBS ${PIDX_GENERIC_RST_LINK_LIBS} ${ZFP_LIBRARIES})
ENDIF()

IF (PMT_FOUND)
  MESSAGE("Configuring pidx with PMT support")
  INCLUDE_DIRECTORIES(${PMT_INSITU_INCLUDE_DIR})
  SET(PIDX_GENERIC_RST_LINK_LIBS ${PIDX_GENERIC_RST_LINK_LIBS} ${PMT_INSITU_LINK_LIBRARIES})
ENDIF()

IF (VTK_RENDERER_FOUND)
  INCLUDE_DIRECTORIES(${VTK_RENDERER_INCLUDE_DIR})
ENDIF()

# ////////////////////////////////////////
# library
# ////////////////////////////////////////

PIDX_ADD_LIBRARY(pidx_generic_rst "${PIDX_GENERIC_RST_SOURCES}")
TARGET_LINK_LIBRARIES(pidx_generic_rst ${OS_SPECIFIC_LIBS} ${PIDX_GENERIC_RST_LINK_LIBS})
SET_TARGET_PROPERTIES(pidx_generic_rst PROPERTIES LINKER_LANGUAGE CXX)
